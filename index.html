#!/usr/bin/env python3 # -*- coding: utf-8 -*- import os import io import base64 from flask import Flask, request, render_template_string from PIL import Image, ImageFilter import pytesseract # Optionally allow configuring Tesseract binary path via env var # e.g., export TESSERACT_CMD=/usr/local/bin/tesseract tesseract_cmd = os.environ.get("TESSERACT_CMD") if tesseract_cmd: pytesseract.pytesseract.tesseract_cmd = tesseract_cmd app = Flask(__name__) HTML = """ CAPTCHA OCR Decoder

CAPTCHA OCR Decoder

Upload a CAPTCHA image to extract its text using Tesseract OCR.

{% if error %}

{{ error }}

{% endif %} {% if text is not none %}
{% if original_data_uri %}

Original image

{% endif %} {% if processed_data_uri %}

Preprocessed for OCR

{% endif %}

Decoded text

{{ text if text.strip() else "(no text detected)" }}
{% endif %}
Tip: High-contrast images with minimal noise yield better OCR. This demo uses single-line mode and a whitelist for alphanumerics.
""" def preprocess_for_captcha(img: Image.Image) -> Image.Image: # Convert to grayscale g = img.convert("L") # Light denoise g = g.filter(ImageFilter.MedianFilter(size=3)) # Simple thresholding to high-contrast B/W # Adjust threshold if needed (120â€“170 range often works for CAPTCHAs) threshold = 140 bw = g.point(lambda p: 255 if p > threshold else 0, mode='1') # Convert back to 8-bit to ensure compatibility with tesseract bw = bw.convert("L") return bw def to_data_uri(img: Image.Image, fmt="PNG") -> str: buf = io.BytesIO() img.save(buf, format=fmt) b64 = base64.b64encode(buf.getvalue()).decode("ascii") return f"data:image/{fmt.lower()};base64,{b64}" @app.route("/", methods=["GET", "POST"]) def index(): error = None text = None original_data_uri = None processed_data_uri = None if request.method == "POST": file = request.files.get("image") if not file or file.filename == "": error = "Please choose an image file." else: if not (file.mimetype or "").startswith("image/"): error = "Unsupported file type. Please upload an image." else: try: raw = file.read() original_img = Image.open(io.BytesIO(raw)).convert("RGB") processed_img = preprocess_for_captcha(original_img) # Build data URIs for previews original_data_uri = to_data_uri(original_img, "PNG") processed_data_uri = to_data_uri(processed_img, "PNG") # Tesseract config: assume a single line of alphanumeric text config = ( "--oem 3 --psm 7 " "-c tessedit_char_whitelist=ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789" ) text = pytesseract.image_to_string(processed_img, config=config) except Exception as e: error = f"Failed to process image: {e}" return render_template_string( HTML, error=error, text=text, original_data_uri=original_data_uri, processed_data_uri=processed_data_uri, ) if __name__ == "__main__": port = int(os.environ.get("PORT", "8000")) app.run(host="0.0.0.0", port=port, debug=True)
